### 1. 物品购买金币检查与扣除
**文件**: `backend/src/routes/items.js`
**位置**: 第 25 行附近（购买接口）

**现状问题**:
- 购买物品时没有检查用户金币是否足够
- 没有扣除用户金币的逻辑
- 存在 TODO 注释但未实现

**需求描述**:
- 用户购买物品前，检查金币余额是否足够
- 购买成功后扣除对应金币
- 金币不足时返回明确错误信息

**实现方法**:
```javascript
// 获取用户金币
const user = db.prepare('SELECT gold FROM users WHERE id = ?').get(req.user.userId);
if (user.gold < totalCost) {
  return res.status(400).json({ error: '金币不足' });
}
// 扣除金币
db.prepare('UPDATE users SET gold = gold - ? WHERE id = ?').run(totalCost, req.user.userId);
```

### 6. 成就自动触发检查
**文件**: `backend/src/routes/achievements.js`

**现状问题**:
- 成就数据存在但不会自动检查完成状态
- 需要手动触发

**需求描述**:
- 在关键操作（升级、战斗胜利、作业完成）后自动检查成就
- 完成后发放奖励（经验、金币、物品）
- 记录成就完成时间

**实现方法**:
```javascript
// 创建成就检查函数
function checkAndAwardAchievement(userId, achievementType, currentValue) {
  const achievement = db.prepare('SELECT * FROM achievements WHERE type = ? AND requirement <= ?').get(achievementType, currentValue);

  // 检查是否已完成
  const existing = db.prepare('SELECT * FROM user_achievements WHERE user_id = ? AND achievement_id = ?').get(userId, achievement.id);
  if (existing) return;

  // 插入成就记录
  db.prepare('INSERT INTO user_achievements (user_id, achievement_id, completed_at) VALUES (?, ?, CURRENT_TIMESTAMP)').run(userId, achievement.id);

  // 发放奖励
  if (achievement.reward_type === 'exp') {
    db.prepare('UPDATE pets SET exp = exp + ? WHERE user_id = ?').run(achievement.reward_value, userId);
  }
}
```

### 9. 装备套装效果
**文件**: `backend/src/routes/equipment.js`

**需求描述**:
- 相同套装的装备有套装属性
- 2 件套 +5% 属性，4 件套 +15% 属性

**实现方法**:
```javascript
// 装备表增加套装字段
db.prepare('ALTER TABLE equipment ADD COLUMN set_name TEXT');

// 计算套装效果
function calculateSetBonus(userId) {
  const equipped = db.prepare('SELECT e.set_name FROM user_equipment ue JOIN equipment e ON ue.equipment_id = e.id WHERE ue.user_id = ? AND ue.equipped = 1').all(userId);
  const setCounts = {};
  equipped.forEach(e => {
    setCounts[e.set_name] = (setCounts[e.set_name] || 0) + 1;
  });

  let totalBonus = { attack: 0, defense: 0, speed: 0 };
  for (const [setName, count] of Object.entries(setCounts)) {
    if (count >= 2) { totalBonus.attack += 5; }
    if (count >= 4) { totalBonus.attack += 15; }
  }
  return totalBonus;
}
```

### 10. 宠物转生系统
**文件**: `backend/src/routes/pets.js`

**需求描述**:
- 消耗"转生丹"进行转生
- 转生后等级回到 1，经验清零
- 基础属性永久提升（如 ×1.2 倍）
- 保留成长阶段

**实现方法**:
```javascript
router.post('/rebirth', authenticateToken, (req, res) => {
  const { item_id } = req.body;

  // 检查转生丹
  const userItem = db.prepare('SELECT * FROM user_items WHERE user_id = ? AND item_id = ? AND quantity > 0').get(req.user.userId, item_id);
  if (!userItem) {
    return res.status(400).json({ error: '没有转生丹' });
  }

  const pet = db.prepare('SELECT * FROM pets WHERE user_id = ?').get(req.user.userId);

  // 转生：等级=1，经验=0，属性提升
  const multiplier = 1.2;
  db.prepare('UPDATE pets SET level = 1, exp = 0, attack = FLOOR(attack * ?), defense = FLOOR(defense * ?), speed = FLOOR(speed * ?) WHERE user_id = ?').run(multiplier, multiplier, multiplier, req.user.userId);

  // 扣除道具
  db.prepare('UPDATE user_items SET quantity = quantity - 1 WHERE user_id = ? AND item_id = ?').run(req.user.userId, item_id);

  res.json({ message: '转生成功，属性永久提升20%' });
});
```
